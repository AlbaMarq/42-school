	char **allocate_memory(int rows, int cols) {
		char **array = malloc(rows * sizeof(char *));
		if(array == NULL) {
			// handle error
			return NULL;
		}

		for(int i = 0; i < rows; i++) {
			array[i] = malloc(cols * sizeof(char));
			if(array[i] == NULL) {
				// handle error
				return NULL;
			}
		}

		return array;
	}
